178  }
179  else if constexpr(avnd::vbox_layout<Item>)

181  subLayout(item, new score::GraphicsVBoxLayout{layout});
182  }
183  else if constexpr(avnd::split_layout<Item>)

185  subLayout(item, new score::GraphicsSplitLayout{layout});

190  new_l->setColumns(Item::columns());

192  }
