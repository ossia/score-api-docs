131  static void
132  run(float freq,
133  float ampl,
134  float ampl_fine,
135  float offset,
136  float offset_fine,
137  float jitter,
138  float custom_phase,
139  const std::string& type,
140  float quantif,
141  ossia::value_port& out,
142  ossia::token_request tk,
143  ossia::exec_state_facade st,
144  State& s)
145  {
146  constexpr const double sine_ratio
147  = ossia::two_pi / ossia::flicks_per_second<double>;
148  const auto& waveform_map = Control::Widgets::waveformMap();
149  const auto elapsed = tk.model_read_duration().impl;
150 
151  if (quantif)
152  {
153  // Determine the frequency with the quantification
154  if (tk.unexpected_bar_change())
155  {
156  s.phase = 0;
157  }
158 
159  // If quantif == 1, we quantize to the bar
160  // => f = 0.5 hz
161  // If quantif == 1/4, we quantize to the quarter
162  // => f = 2hz
163  // -> sin(elapsed * freq * 2 * pi / fps)
164  // -> sin(elapsed * 4 * 2 * pi / fps)
165  freq = 1. / (2. * quantif);
166  }
167 
168  const auto ph_delta = elapsed * freq * sine_ratio;
169 
170  if (const auto it = waveform_map.find_key(type))
171  {
172  auto ph = s.phase;
173  if (jitter > 0)
174  {
175  ph += std::normal_distribution<float>(0., 0.25)(s.rd) * jitter;
176  }
177 
178  ampl += ampl_fine;
179  offset += offset_fine;
180 
181  using namespace Control::Widgets;
182 
183  const auto add_val = [&](auto new_val) {
184  const auto [tick_start, d] = st.timings(tk);
185  out.write_value(ampl * new_val + offset, tick_start);
186  };
187  switch (*it)
188  {
189  case Sin:
190  add_val(std::sin(custom_phase + ph));
191  break;
192  case Triangle:
193  add_val(std::asin(std::sin(custom_phase + ph)) / ossia::half_pi);
194  break;
195  case Saw:
196  add_val(std::atan(std::tan(custom_phase + ph)) / ossia::half_pi);
197  break;
198  case Square:
199  add_val((std::sin(custom_phase + ph) > 0.f) ? 1.f : -1.f);
200  break;
201  case SampleAndHold:
202  {
203  const auto start_s = std::sin(custom_phase + ph);
204  const auto end_s = std::sin(custom_phase + ph + ph_delta);
205  if ((start_s > 0 && end_s <= 0) || (start_s <= 0 && end_s > 0))
206  {
207  add_val(std::uniform_real_distribution<float>(-1.f, 1.f)(s.rd));
208  }
209  break;
210  }
211  case Noise1:
212  add_val(std::uniform_real_distribution<float>(-1.f, 1.f)(s.rd));
213  break;
214  case Noise2:
215  add_val(std::normal_distribution<float>(0.f, 1.f)(s.rd));
216  break;
217  case Noise3:
218  add_val(std::clamp(
219  std::cauchy_distribution<float>(0.f, 1.f)(s.rd), 0.f, 1.f));
220  break;
221  }
222  }
223 
224  s.phase += ph_delta;
225  }
