49  void operator()(const halp::tick_flicks& tk)
50  {
51  constexpr const double sine_ratio = ossia::two_pi / ossia::flicks_per_second<double>;
52  const auto elapsed = tk.model_read_duration();
53 
54  const auto quantif = inputs.quant.value;
55  auto freq = inputs.freq.value;
56  auto ampl = inputs.ampl.value;
57  auto offset = inputs.offset.value;
58  const auto jitter = inputs.jitter.value;
59  auto custom_phase = inputs.phase.value;
60  const auto type = inputs.waveform.value;
61  if(quantif)
62  {
63  // Determine the frequency with the quantification
64  if(tk.unexpected_bar_change())
65  {
66  this->phase = 0;
67  }
68 
69  // If quantif == 1, we quantize to the bar
70  // => f = 0.5 hz
71  // If quantif == 1/4, we quantize to the quarter
72  // => f = 2hz
73  // -> sin(elapsed * freq * 2 * pi / fps)
74  // -> sin(elapsed * 4 * 2 * pi / fps)
75  freq = 1. / (2. * quantif);
76  }
77 
78  const auto ph_delta = elapsed * freq * sine_ratio;
79 
80  {
81  auto ph = this->phase;
82  if(jitter > 0)
83  {
84  ph += std::normal_distribution<float>(0., 0.25)(this->rd) * jitter;
85  }
86 
87  using namespace Control::Widgets;
88 
89  const auto add_val
90  = [&](auto new_val) { outputs.out.value = ampl * new_val + offset; };
91  switch(type)
92  {
93  case Sin:
94  add_val(std::sin(custom_phase + ph));
95  break;
96  case Triangle:
97  add_val(std::asin(std::sin(custom_phase + ph)) / ossia::half_pi);
98  break;
99  case Saw:
100  add_val(std::atan(std::tan(custom_phase + ph)) / ossia::half_pi);
101  break;
102  case Square:
103  add_val((std::sin(custom_phase + ph) > 0.f) ? 1.f : -1.f);
104  break;
105  case SampleAndHold: {
106  const auto start_s = std::sin(custom_phase + ph);
107  const auto end_s = std::sin(custom_phase + ph + ph_delta);
108  if((start_s > 0 && end_s <= 0) || (start_s <= 0 && end_s > 0))
109  {
110  add_val(std::uniform_real_distribution<float>(-1.f, 1.f)(this->rd));
111  }
112  break;
113  }
114  case Noise1:
115  add_val(std::uniform_real_distribution<float>(-1.f, 1.f)(this->rd));
116  break;
117  case Noise2:
118  add_val(std::normal_distribution<float>(0.f, 1.f)(this->rd));
119  break;
120  case Noise3:
121  add_val(
122  std::clamp(std::cauchy_distribution<float>(0.f, 1.f)(this->rd), 0.f, 1.f));
123  break;
124  }
125  }
126 
127  this->phase += ph_delta;
128  }
