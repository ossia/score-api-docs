137  QObject::connect(pressed, &QState::entered, [&]() {

154  if (!prev_csts.empty())
155  {
156  // We find the one that starts the latest.
157  TimeVal t = TimeVal::zero();
158  for (const auto& cst_id : prev_csts)

160  const auto& other_date = scenar.interval(cst_id).date();
161  if (other_date > t)
162  t = other_date;

173

177  QObject::connect(onlyMoving, &QState::entered, [&]() {
178  auto& scenar = stateMachine.model();
179  // If we came here through a state.
180  auto evId = this->clickedEvent;
181  if (!bool(evId) && bool(this->clickedState))

183  evId = scenar.state(*this->clickedState).eventId();

256  void merge(const StateModel& st, TimeVal date)
257  {
258  auto& ev = Scenario::parentEvent(st, this->m_scenario);
259  auto& ts = Scenario::parentTimeSync(ev, this->m_scenario);
260 
261  TimeSyncPresenter& sts_pres = m_sm.presenter().timeSync(ts.id());
262 
263  std::vector<QGraphicsItem*> toIgnore;
264  toIgnore.push_back(sts_pres.view());
265  toIgnore.push_back(&sts_pres.trigger());
266  for(auto& ev_id : sts_pres.model().events())
267  {
268  auto& ev = m_sm.presenter().event(ev_id);
269  toIgnore.push_back(ev.view());
270  for(auto& st_id : ev.model().states())
271  {
272  auto& st = m_sm.presenter().state(st_id);
273  toIgnore.push_back(st.view());
274  }
275  }
276 
277  for(auto& itv : Scenario::previousIntervals(sts_pres.model(), this->m_scenario))
278  {
279  auto& sst_pres = m_sm.presenter().state(this->m_scenario.interval(itv).startState());
280  auto& sev_pres = m_sm.presenter().event(sst_pres.model().eventId());
281  TimeSyncPresenter& sts_pres = m_sm.presenter().timeSync(sev_pres.model().timeSync());
282 
283  toIgnore.push_back(sts_pres.view());
284  for(auto& ev : sts_pres.events())
285  {
286  toIgnore.push_back(ev->view());
287  for(auto& st : ev->states())
288  {
289  toIgnore.push_back(st->view());
290  }
291  }
292 
293  toIgnore.push_back(&sts_pres.trigger());
294  }
295 
296  for(auto& itv : Scenario::nextIntervals(sts_pres.model(), this->m_scenario))
297  {
298  auto& sst_pres = m_sm.presenter().state(this->m_scenario.interval(itv).endState());
299  auto& sev_pres = m_sm.presenter().event(sst_pres.model().eventId());
300  auto& sts_pres = m_sm.presenter().timeSync(sev_pres.model().timeSync());
301 
302  toIgnore.push_back(sts_pres.view());
303  for(auto& ev : sts_pres.events())
304  {
305  toIgnore.push_back(ev->view());
306  for(auto& st : ev->states())
307  {
308  toIgnore.push_back(st->view());
309  }
310  }
311 
312  toIgnore.push_back(&sts_pres.trigger());
313  }
314 
315  QGraphicsItem* item = m_sm.itemAt({date, st.heightPercentage()}, toIgnore);
316 
317  if (auto stateToMerge = qgraphicsitem_cast<Scenario::StateView*>(item))
318  {
319  // this->rollback();
320  this->m_movingDispatcher.template submit<Command::MergeEvents>(
321  this->m_scenario,
322  ev.id(),
323  Scenario::parentEvent(stateToMerge->presenter().model().id(), this->m_scenario).id());
324  }
325  else if (auto eventToMerge = qgraphicsitem_cast<Scenario::EventView*>(item))
326  {
327  // this->rollback();
328  this->m_movingDispatcher.template submit<Command::MergeEvents>(
329  this->m_scenario, ev.id(), eventToMerge->presenter().model().id());
330  }
331  else if (auto syncToMerge = qgraphicsitem_cast<Scenario::TimeSyncView*>(item))
332  {
333  // this->rollback();
334  this->m_movingDispatcher.template submit<Command::MergeTimeSyncs>(
335  this->m_scenario, ts.id(), syncToMerge->presenter().model().id());
336  }
337  }
