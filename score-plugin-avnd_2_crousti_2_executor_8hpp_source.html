970  void connect_worker(const ::Execution::Context& ctx, avnd::effect_container<Node>& eff)
971  {
972  if constexpr(avnd::has_worker<Node>)
973  {
974  // Initialize the thread pool beforehand
975  auto& tq = score::TaskPool::instance();
976  using worker_type = decltype(eff.effect.worker);
977  for(auto& eff : eff.effects())
978  {
979  std::weak_ptr eff_ptr = std::shared_ptr<Node>(this->node, &eff);
980  std::weak_ptr qex_ptr = std::shared_ptr<Execution::ExecutionCommandQueue>(
981  ctx.alias.lock(), &ctx.executionQueue);
982 
983  eff.worker.request
984  = [&tq, qex_ptr = std::move(qex_ptr),
985  eff_ptr = std::move(eff_ptr)]<typename... Args>(Args&&... f) mutable {
986  // request() is invoked in the DSP / processor thread
987  // and just posts the task to the thread pool
988  tq.post([eff_ptr = std::move(eff_ptr), qex_ptr = std::move(qex_ptr),
989  ... ff = std::forward<Args>(f)]() mutable {
990  // This happens in the worker thread
991  // If for some reason the object has already been removed, not much
992  // reason to perform the work
993  if(!eff_ptr.lock())
994  return;
995 
996  using type_of_result
997  = decltype(worker_type::work(std::forward<decltype(ff)>(ff)...));
998  if constexpr(std::is_void_v<type_of_result>)
999  {
1000  worker_type::work(std::forward<decltype(ff)>(ff)...);
1001  }
1002  else
1003  {
1004  // If the worker returns a std::function, it
1005  // is to be invoked back in the processor DSP thread
1006  auto res = worker_type::work(std::forward<decltype(ff)>(ff)...);
1007  if(!res)
1008  return;
1009 
1010  // Execution queue is currently spsc from main thread to an exec thread,
1011  // we cannot just yeet the result back from the thread-pool
1012  ossia::qt::run_async(
1013  qApp, [eff_ptr = std::move(eff_ptr), qex_ptr = std::move(qex_ptr),
1014  res = std::move(res)]() mutable {
1015  // Main thread
1016  std::shared_ptr qex = qex_ptr.lock();
1017  if(!qex)
1018  return;
1019 
1020  qex->enqueue(
1021  [eff_ptr = std::move(eff_ptr), res = std::move(res)]() mutable {
1022  // DSP / processor thread
1023  // We need res to be mutable so that the worker can use it to e.g. store
1024  // old data which will be freed back in the main thread
1025  if(auto p = eff_ptr.lock())
1026  res(*p);
1027  });
1028  });
1029  }
1030  });
1031  };
1032  }
1033  }
