110 inline void updatePropagated(
111  const ossia::node_ptr& process_node,
112  const ossia::node_ptr& interval_node,
113  ossia::graph_interface& g,
114  std::size_t port_idx,
115  bool is_propagated) noexcept
116 {
117  const auto& outs = process_node->root_outputs();
118 
119  if (port_idx >= outs.size())
120  return;
121 
122  const ossia::outlet& outlet = *outs[port_idx];
123 
124  if (!outlet.target<ossia::audio_port>())
125  return;
126 
127  // Remove cables if depropagated, add cables if repropagated
128  if (is_propagated)
129  {
130  for (const ossia::graph_edge* edge : outlet.targets)
131  {
132  if (edge->in_node == interval_node)
133  return;
134  }
135 
136  auto cable = ossia::make_edge(
137  ossia::immediate_glutton_connection{},
138  outs[port_idx],
139  interval_node->root_inputs()[0],
140  process_node,
141  interval_node);
142  g.connect(cable);
143  }
144  else
145  {
146  for (ossia::graph_edge* edge : outlet.targets)
147  {
148  if (edge->in_node == interval_node)
