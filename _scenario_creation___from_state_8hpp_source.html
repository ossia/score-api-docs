261  template <typename Fun>
262  void creationCheck(Fun&& fun)
263  {
264  const auto& scenar = this->m_parentSM.model();
265  if (!this->clickedState)
266  return;
267  Scenario::EditionSettings& settings = this->m_parentSM.editionSettings();
268  const bool sequence = settings.tool() == Tool::CreateSequence;
269  const bool new_event = qApp->keyboardModifiers() & Qt::ALT;
270  const bool graph_interval = settings.tool() == Scenario::Tool::CreateGraph;
271  auto& st = scenar.state(*this->clickedState);
272  auto& ev = Scenario::parentEvent(st, scenar);
273  if(ev.date() > this->currentPoint.date && !graph_interval)
274  return;
275 
276  if (new_event && !sequence)
277  {
278  // Create new event on the timesync
279  auto tn = ev.timeSync();
280  auto cmd
281  = new Scenario::Command::CreateEvent_State{this->m_scenario, tn, this->currentPoint.y};
282  this->m_dispatcher.submit(cmd);
283 
284  this->createdEvents.append(cmd->createdEvent());
285  this->createdStates.append(cmd->createdState());
286  fun(this->createdStates.first());
287  }
288  else
289  {
290  if (!sequence)
291  {
292  // Create new state on the event
293  auto cmd = new Scenario::Command::CreateState{
294  this->m_scenario, st.eventId(), this->currentPoint.y};
295  this->m_dispatcher.submit(cmd);
296 
297  this->createdStates.append(cmd->createdState());
298  fun(this->createdStates.first());
299  }
300  else
301  {
302  if (!st.nextInterval()) // TODO & deltaY < deltaX
303  {
304  this->currentPoint.y = st.heightPercentage();
305  fun(*this->clickedState);
306  }
307  else
308  {
309  // Create new state on the event
310  auto cmd = new Scenario::Command::CreateState{
311  this->m_scenario, st.eventId(), this->currentPoint.y};
312  this->m_dispatcher.submit(cmd);
313 
314  this->createdStates.append(cmd->createdState());
315  fun(this->createdStates.first());
316  // create a single state on the same event (deltaY > deltaX)
317  }
318  }
319  }
320  }
