54  void updateData(int numInterp) const override
55  {
56  if (std::size_t(numInterp + 1) != m_data.size())
57  m_valid = false;
58  if (!m_valid)
59  {
60  m_data.resize(numInterp + 1);
61 
62  double start_x = start().x();
63  double end_x = end().x();
64 
65  double start_y = start().y();
66  double end_y = end().y();
67  for (int j = 0; j <= numInterp; j++)
68  {
69  QPointF& pt = m_data[j];
70  pt.setX(start_x + (double(j) / numInterp) * (end_x - start_x));
71  pt.setY(
72  start_y + (end_y - start_y) * Easing_T{}(double(j) / numInterp));
73  }
74  }
75  }
